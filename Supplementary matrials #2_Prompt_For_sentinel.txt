    - High-power devices (e.g., AC, purifier) are running unnecessarily — assess if they're still needed.
    - No user interaction has occurred for an extended time — check if default operation is wasting energy.

You are given the following inputs for judgment:
1. Current and previous environmental data:

Task Output:
You must respond **using Function-Calling only**, with the following format:

def sentinel_judge(state: dict, memory: list = None) -> dict:
    room = state.get("room", "Office")
    power_summary = get_room_power_summary(room)
    users = users_in_room(room)
    user_info_lines = []
    for u in users:
        info = f"- {u['name']}"
        extras = []
        if u.get("gender"): extras.append(f"{u['gender']}")
        if u.get("age"): extras.append(f"{u['age']}yrs")
        if u.get("height"): extras.append(f"{u['height']}cm")
        if u.get("weight"): extras.append(f"{u['weight']}kg")
        if u.get("conditions"): extras.append(f"Chronic health conditions: {', '.join(u['conditions'])}")
        if extras:
            info += "（" + "，".join(extras) + "）"
        user_info_lines.append(info)

    user_info_block = "\nOccupants Present:\n" + "\n".join(user_info_lines) if user_info_lines else "\nNo Occupant presence detected."

    try:
        current_time = time.strftime("%Y-%m-%d %H:%M:%S", time.localtime())
        messages = [
            {"role": "user", "content":
                f"Current Time:{current_time}\n\n"

                f"The following is the current environmental state (including recent user intent):\n{json.dumps(state, ensure_ascii=False, indent=2)}\n\n"
                f"Here are the last 5 sentinel memory records:\n{format_sentinel_memory(memory)}\n\n"
                f"Based on the most recent user input (e.g., the 'recent_user_intent' field), please determine whether there is anything in the current environment that warrants notifying the user.\n"
                f"If the user’s recent expression appears inconsistent with current environmental conditions, please also provide a recommendation or alert."

             }
